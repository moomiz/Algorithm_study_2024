

### 1. 문제 출처
   https://www.acmicpc.net/problem/16236

### 2. input 및 output

`input`
 - N : 공간의 크기
 - N개의 줄에 공간의 상태가 주어짐
	 - 0 : 빈 칸
	 - 1 ~ 6 : 물고기의 크기
	 - 9 : 아기 상어의 위치

`output`
 - 아기 상어가 엄마 상어에게 도움을 요청하지 않고 물고기를 잡아먹을 수 있는 시간

#### c++

```c++
#include <iostream>
#include <queue>
#include <cstring>
#include <climits>

using namespace std;

int n, ans, cnt;
int map[22][22];
int dist[22][22];
int result[3];
int sr, sc, sz;
int dx[] = {0, 0, -1, 1};
int dy[] = {1, -1, 0, 0};

struct Position
{
    int r, c;
};

int main()
{
    cin >> n;
    for (int i = 0; i < n; i++)
    {
        for (int j = 0; j < n; j++)
        {
            cin >> map[i][j];
            if (map[i][j] == 9)
            {
                sr = i;
                sc = j;
                sz = 2;
                map[i][j] = 0;
            }
        }
    }

    ans = 0;
    cnt = 0;
    while (true)
    {
        find(); // 먹을 수 있는 물고기 찾기
        if (result[2] == INT_MAX) // 잡을 수 있는 물고기가 없으면
        {
            cout << ans << endl;
            break;
        }
        else
        {
            sr = result[0];
            sc = result[1];
            ans += result[2];

            map[sr][sc] = 0; // 먹은 물고기 제거

            cnt += 1;
            if (cnt == sz) // 먹은 개수가 상어 크기와 같으면 상어 레벨업
            {
                sz += 1;
                cnt = 0;
            }
        }
    }

    return 0;
}

void find() 
{
    result[0] = 0;
    result[1] = 0;
    result[2] = INT_MAX;

    bfs(); // 물고기 맵의 최소거리 업데이트
    int r = 0, c = 0, min = INT_MAX;

    for (int i = 0; i < n; i++) // 조건에 맞게 우선 순위 탐색하면서 최소거리 구하기
    {
        for (int j = 0; j < n; j++)
        {
            if (dist[i][j] != -1 && map[i][j] > 0 && map[i][j] < sz)
            {
                if (dist[i][j] < min)
                {
                    r = i;
                    c = j;
                    min = dist[i][j];
                }
            }
        }
    }
    result[0] = r;
    result[1] = c;
    result[2] = min;
}

void bfs()
{
    memset(dist, -1, sizeof(dist));
    queue<Position> q;
    q.push({sr, sc});
    dist[sr][sc] = 0;

    while (!q.empty())
    {
        Position p = q.front();
        q.pop();
        for (int d = 0; d < 4; d++)
        {
            int nr = p.r + dx[d];
            int nc = p.c + dy[d];
            if (nr >= 0 && nr < n && nc >= 0 && nc < n && map[nr][nc] <= sz && dist[nr][nc] == -1)
            {
                dist[nr][nc] = dist[p.r][p.c] + 1;
                q.push({nr, nc});
            }
        }
    }
}
```
